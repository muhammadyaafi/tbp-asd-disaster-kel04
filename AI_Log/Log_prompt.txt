class Queue:
    class _Node:
        """
        Node ringan untuk Linked List (Goodrich §7.1, hal. 256).
        Menggunakan __slots__ untuk menghemat memori.
        """
        __slots__ = '_element', '_next'
        def __init__(self, data):
            self._element = data
            self._next = None

    def __init__(self):
        self._head = None
        self._tail = None
        self._size = 0

    def is_empty(self):
        return self._size == 0

    def enqueue(self, e):
        newest = self._Node(e, None)        # node baru, next=None (akan jadi tail)
        if self.is_empty():
            self._head = newest             # queue sebelumnya kosong, head = node baru
        else:
            self._tail._next = newest       # hubungkan tail lama ke node baru
        self._tail = newest                 # update tail ke node baru
        self._size += 1

    def dequeue(self):
        if self.is_empty():
            raise Empty('Queue is empty')
        answer = self._head._element
        self._head = self._head._next   # majukan head
        self._size -= 1
        if self.is_empty():
            self._tail = None   # queue kosong, tail tidak valid lagi
        return answer

class Stack:
    """Stack berbasis Singly Linked List, LIFO."""
    def __init__(self, kapasitas=None):
        self.top = None      # Menunjuk ke elemen teratas [2, 4]
        self._size = 0       # Melacak jumlah elemen [2, 5]
        self.kapasitas = kapasitas # Opsional: untuk Topik 3 (kapasitas 10) atau Topik 4 (kapasitas 20) [6, 7]

    def push(self, data) -> bool:
        """Big-O waktu: O(1) - Menambahkan elemen ke posisi teratas [8, 9]."""
        # Cek jika ada batasan kapasitas (seperti pada Topik 3 atau 4) [6, 10]
        if self.kapasitas is not None and self._size >= self.kapasitas:
            return False 
        
        new_node = LLNode(data)
        new_node.next = self.top # Node baru menunjuk ke top lama
        self.top = new_node      # Top sekarang adalah node baru [5]
        self._size += 1
        return True

    def pop(self):
        """Big-O waktu: O(1) - Mengambil dan menghapus elemen teratas [8, 9]."""
        if self.is_empty():
            return None
        
        popped_node = self.top
        self.top = self.top.next # Geser top ke node di bawahnya [5]
        self._size -= 1
        return popped_node.data

    def peek(self):
        """Big-O waktu: O(1) - Melihat elemen teratas tanpa menghapusnya [2, 5]."""
        return self.top.data if self.top else None

    def is_empty(self) -> bool:
        """Mengecek apakah stack kosong."""
        return self._size == 0

    def __len__(self) -> int:
        """Mengembalikan ukuran stack saat ini."""
        return self._size

2. Implementasi LinkedStack
Metode ini menggunakan konsep nested class _Node untuk menyimpan elemen dan referensi ke node berikutnya
. Top dari stack selalu berada di bagian head dari linked list agar operasi penambahan dan penghapusan dapat dilakukan dalam waktu konstan O(1)
.
class LinkedStack:
    """LIFO Stack implementation using a singly linked list for storage."""

    #-------------------------- nested Node class --------------------------
    class _Node:
        """Lightweight, nonpublic class for storing a singly linked node."""
        __slots__ = '_element', '_next' # streamline memory usage

        def __init__(self, element, next): # initialize node’s fields
            self._element = element # reference to user’s element
            self._next = next       # reference to next node

    #------------------------------- stack methods -------------------------------
    def __init__(self):
        """Create an empty stack."""
        self._head = None           # reference to the head node
        self._size = 0              # number of stack elements

    def __len__(self):
        """Return the number of elements in the stack."""
        return self._size

    def is_empty(self):
        """Return True if the stack is empty."""
        return self._size == 0

    def push(self, e):
        """Add element e to the top of the stack."""
        self._head = self._Node(e, self._head) # create and link a new node
        self._size += 1

    def top(self):
        """Return (but do not remove) the element at the top of the stack.
        Raise Empty exception if the stack is empty.
        """
        if self.is_empty():
            raise Empty('Stack is empty')
        return self._head._element  # top of stack is at head of list

    def pop(self):
        """Remove and return the element from the top of the stack (i.e., LIFO).
        Raise Empty exception if the stack is empty.
        """
        if self.is_empty():
            raise Empty('Stack is empty')
        answer = self._head._element
        self._head = self._head._next # bypass the former top node
        self._size -= 1
        return answer

class LinkedStack:
    """Implementasi Stack LIFO menggunakan singly linked list untuk penyimpanan."""

    #-------------------------- Kelas Node Internal --------------------------
    class _Node:
        """Kelas internal ringan untuk menyimpan node linked list satu arah."""
        __slots__ = '_element', '_next' # Mengoptimalkan penggunaan memori [2, 5]

        def __init__(self, element, next): 
            self._element = element # Referensi ke data kiriman bantuan [5]
            self._next = next       # Referensi ke node di bawahnya

    #------------------------------- Metode Stack -------------------------------
    def __init__(self):
        """Membuat stack logistik yang kosong."""
        self._head = None           # Menunjuk ke node teratas (paling baru) [2]
        self._size = 0              # Melacak jumlah log pengiriman

    def __len__(self):
        """Mengembalikan jumlah data pengiriman dalam stack."""
        return self._size

    def is_empty(self):
        """Mengembalikan True jika belum ada pengiriman yang dicatat."""
        return self._size == 0

    def push(self, e):
        """Menambahkan data pengiriman baru ke bagian teratas stack."""
        # Buat node baru dan hubungkan ke head yang lama
        self._head = self._Node(e, self._head) 
        self._size += 1

    def top(self):
        """Melihat data pengiriman teratas tanpa menghapusnya.
        Memunculkan pengecualian Empty jika stack kosong.
        """
        if self.is_empty():
            raise Empty('Stack kosong: Tidak ada log yang bisa dilihat')
        return self._head._element  # Bagian head adalah posisi teratas stack [2]

    def pop(self):
        """Menghapus dan mengembalikan data pengiriman paling terakhir (LIFO).
        Sangat berguna untuk fitur ROLLBACK pada sistem logistik bencana. [4]
        """
        if self.is_empty():
            raise Empty('Stack kosong: Tidak ada pengiriman untuk dibatalkan')
        
        answer = self._head._element
        self._head = self._head._next # Menggeser head ke node sebelumnya [3]
        self._size -= 1
        return answer

class Stack: 
    def __init__(self): 
        # Inisialisasi stack kosong sesuai standar buku [2]
        self.top = None # Buku menggunakan istilah '_head'
        self._size = 0 
    
    def is_empty(self):
        """Mengembalikan True jika stack kosong [2]."""
        return self._size == 0

    def push(self, data): 
        """Big-O: O(1) - Menambahkan elemen di posisi teratas [2, 3]."""
        # Membuat node baru dan menghubungkannya ke top yang lama
        new_node = LLNode(data)
        new_node.next = self.top
        self.top = new_node
        self._size += 1 
    
    def pop(self): 
        """Big-O: O(1) - Menghapus dan mengambil elemen teratas [3, 4]."""
        if self.is_empty():
            return None # Menghindari error jika stack kosong
        
        answer = self.top.data
        self.top = self.top.next # Menggeser top ke node di bawahnya
        self._size -= 1
        return answer 
    
    def peek(self): 
        """Melihat elemen teratas tanpa menghapusnya (sama dengan top() di buku) [2]."""
        return self.top.data if self.top else None 
    
    def to_list(self): 
        """Mengonversi stack menjadi list untuk keperluan log pengiriman [5]."""
        # Melakukan traversal (pointer hopping) dari top ke bottom
        result = []
        curr = self.top
        while curr is not None:
            result.append(curr.data)
            curr = curr.next
        return result

    def __len__(self):
        return self._size

class Stack: 
    def __init__(self): 
        # Inisialisasi stack kosong sesuai prinsip Linked List di buku
        self.top = None # 'top' menunjuk ke node teratas (sama dengan '_head' di buku) [2]
        self._size = 0 
    
    def is_empty(self) -> bool:
        """Mengembalikan True jika stack logistik kosong."""
        return self._size == 0

    def push(self, data): 
        """
        Big-O: O(1).
        Menambahkan riwayat pengiriman baru ke posisi teratas stack. [2]
        """
        # Membuat node baru menggunakan LLNode yang sudah didefinisikan temanmu
        new_node = LLNode(data)
        # Menghubungkan node baru ke node yang sebelumnya ada di posisi top
        new_node.next = self.top
        # Memperbarui posisi top ke node yang baru saja dimasukkan
        self.top = new_node
        self._size += 1 
    
    def pop(self): 
        """
        Big-O: O(1).
        Mengambil dan menghapus log terakhir. Digunakan untuk fitur ROLLBACK. [4]
        """
        if self.is_empty():
            return None
        
        # Menyimpan data yang akan diambil
        answer = self.top.data
        # Menggeser penunjuk top ke node di bawahnya (menghapus referensi node teratas) [4]
        self.top = self.top.next
        self._size -= 1
        return answer 
    
    def peek(self): 
        """
        Big-O: O(1).
        Melihat data pengiriman teratas tanpa menghapusnya.
        """
        return self.top.data if self.top else None 
    
    def to_list(self): 
        """
        Big-O: O(n).
        Menelusuri seluruh stack dan mengembalikannya dalam bentuk list.
        Sangat berguna untuk fitur LOG_PENGIRIMAN (menampilkan riwayat). [5]
        """
        result = []
        # Melakukan 'link hopping' atau traversal dari top ke bottom [5]
        curr = self.top
        while curr is not None:
            result.append(curr.data)
            curr = curr.next
        return result

    def __len__(self):
        """Mengembalikan jumlah elemen dalam stack."""
        return self._size

class BSTLokasi:
    def __init__(self):
        self.root = None

    def insert(self, lokasi: Lokasi):
        if self.root is None:
            self.root = BSTNodeLok(lokasi)
        else:
            self._insert_recursive(self.root, lokasi)

    def _insert_recursive(self, node, lokasi):
        if lokasi.kode < node.lokasi.kode:
            if node.left is None:
                node.left = BSTNodeLok(lokasi)
            else:
                self._insert_recursive(node.left, lokasi)
        elif lokasi.kode > node.lokasi.kode:
            if node.right is None:
                node.right = BSTNodeLok(lokasi)
            else:
                self._insert_recursive(node.right, lokasi)

    def search(self, kode: str) -> Optional[Lokasi]:
        curr = self.root
        while curr:
            if kode == curr.lokasi.kode:
                return curr.lokasi
            curr = curr.left if kode < curr.lokasi.kode else curr.right
        return None

class PriorityQueueBantuan:
    """Lokasi KRITIS (level=1) selalu dilayani lebih dulu."""
    def __init__(self):
        self.head = None
        self._size = 0

    def enqueue(self, bantuan):
        """
        Big-O: O(n) - Mencari posisi yang tepat berdasarkan level prioritas.
        Prioritas lebih kecil (Level 1) diletakkan di depan.
        """
        new_node = LLNode(bantuan)
        
        # Kasus 1: Antrian kosong atau prioritas bantuan baru lebih tinggi (angka level lebih kecil) dari Head
        if self.head is None or bantuan.prioritas < self.head.data.prioritas:
            new_node.next = self.head
            self.head = new_node
        else:
            # Kasus 2: Cari posisi yang tepat di tengah atau akhir antrian
            # Menggunakan logika <= agar bantuan dengan prioritas sama tetap mempertahankan urutan FIFO (masuk belakangan di posisi belakang)
            current = self.head
            while current.next is not None and current.next.data.prioritas <= bantuan.prioritas:
                current = current.next
            
            new_node.next = current.next
            current.next = new_node
        
        self._size += 1

    def dequeue(self) -> Optional[Bantuan]:
        """
        Big-O: O(1) - Selalu mengambil elemen dari Head.
        """
        if self.head is None:
            return None
        
        # Ambil data dari head
        bantuan_diproses = self.head.data
        # Geser head ke node berikutnya
        self.head = self.head.next
        self._size -= 1
        return bantuan_diproses

    def __len__(self):
        """Mengembalikan jumlah elemen dalam antrian."""
        return self._size

    def is_empty(self):
        """Mengecek apakah antrian kosong."""
        return self._size == 0

Operasi enqueue (O(n)): Karena ini adalah Priority Queue berbasis Linked List terurut, setiap kali ada bantuan baru masuk melalui perintah KIRIM, sistem akan menelusuri daftar untuk menempatkannya di posisi yang sesuai dengan level bencananya
. Bantuan dengan level KRITIS (1) akan selalu berada di depan bantuan level SEDANG (2) atau RINGAN (3)
.
Operasi dequeue (O(1)): Karena daftar sudah dalam keadaan terurut saat proses insertion, maka saat PROSES_BANTUAN dipanggil, kita cukup mengambil elemen di bagian paling depan (head). Hal ini menjamin efisiensi waktu pengambilan data yang sangat cepat
.
Tie-break FIFO: Pada bagian current.next.data.prioritas <= bantuan.prioritas, penggunaan operator <= memastikan bahwa jika ada dua permintaan bantuan dengan level prioritas yang sama, bantuan yang datang lebih dulu akan dilayani lebih dulu (prinsip First-In First-Out)
.
Catatan Tambahan: Pastikan kamu menggunakan objek dari kelas Bantuan saat memanggil fungsi enqueue ini agar atribut .prioritas dapat diakses dengan benar oleh logika di atas
. Jangan lupa untuk mendokumentasikan analisis Big-O ini di dalam laporanmu karena itu merupakan poin penilaian utama

Implementasi Circular Queue (Berbasis Array)
Berbeda dengan Priority Queue yang menggunakan Linked List, Circular Queue menggunakan array dengan kapasitas tetap (misalnya 50 slot) untuk mensimulasikan buffer stok
.
class CircularQueue:
    """
    Circular Queue berbasis array (kapasitas tetap).
    Digunakan untuk buffer FIFO agar produk lama tidak tertimbun.
    """
    def __init__(self, kapasitas):
        self.kapasitas = kapasitas
        self.buffer = [None] * kapasitas
        self.front = 0
        self.rear = 0
        self._size = 0

    def enqueue(self, produk) -> bool:
        """
        Big-O: O(1).
        Menambahkan elemen di posisi rear.
        """
        if self.is_full():
            return False  # Buffer penuh
        
        self.buffer[self.rear] = produk
        # Geser rear secara melingkar menggunakan modulo
        self.rear = (self.rear + 1) % self.kapasitas
        self._size += 1
        return True

    def dequeue(self):
        """
        Big-O: O(1).
        Mengambil elemen tertua di posisi front.
        """
        if self.is_empty():
            return None
        
        produk_keluar = self.buffer[self.front]
        self.buffer[self.front] = None  # Opsional: bersihkan slot
        
        # Geser front secara melingkar menggunakan modulo
        self.front = (self.front + 1) % self.kapasitas
        self._size -= 1
        return produk_keluar

    def is_full(self):
        """Big-O: O(1)."""
        return self._size == self.kapasitas

    def is_empty(self):
        """Big-O: O(1)."""
        return self._size == 0

    def __len__(self):
        return self._size
Poin Penting Perbedaan dengan Priority Queue:
Mekanisme Indeks: Menggunakan operator % (modulo) untuk memastikan saat indeks mencapai akhir array, ia akan kembali ke posisi 0
. Inilah yang membuatnya disebut "Circular".
Efisiensi Waktu (O(1)): Berbeda dengan Priority Queue berbasis Linked List yang enqueue-nya bisa mencapai O(n) karena harus mencari posisi urutan, Circular Queue selalu O(1) untuk tambah dan hapus karena langsung menuju indeks front atau rear
.
Kapasitas Terbatas: Struktur ini memiliki kapasitas tetap (misalnya 50 slot). Jika penuh, data baru tidak bisa masuk kecuali ada data lama yang dikeluarkan (dequeue)
.
Kasus Penggunaan: Di Topik 10, ini digunakan untuk simulasi gudang pangan guna memastikan prinsip First-In First-Out (FIFO) untuk meminimalkan produk kadaluarsa
.
Catatan untuk Topik 9: Jika kamu tetap mengerjakan Topik 9 (Disaster), pastikan kamu menggunakan Priority Queue berbasis Linked List yang sebelumnya kita bahas, karena sistem bencana membutuhkan pengurutan lokasi berdasarkan level KRITIS
. Circular Queue biasanya tidak mendukung prioritas secara langsung tanpa modifikasi tambahan.

Berikut adalah implementasi lengkap program BST Registry Lokasi untuk Topik 9 (Disaster Response Logistics System) sesuai dengan spesifikasi teknis dan starter code yang ada di sumber
:
from dataclasses import dataclass
from typing import Optional, List

# 1. Definisi Struktur Data Lokasi [1]
@dataclass
class Lokasi:
    kode: str          # Kunci utama BST (misal: 'L01')
    nama: str
    level: int         # 1=KRITIS, 2=SEDANG, 3=RINGAN
    populasi: int
    status: int = 0    # 0=BELUM, 1=TERKIRIM

# 2. Implementasi Node BST [2]
class BSTNodeLok:
    def __init__(self, lokasi: Lokasi):
        self.lokasi = lokasi
        self.left: Optional['BSTNodeLok'] = None
        self.right: Optional['BSTNodeLok'] = None

# 3. Implementasi Kelas BST Registry Lokasi [3]
class BSTLokasi:
    def __init__(self):
        self.root: Optional[BSTNodeLok] = None

    def insert(self, lokasi: Lokasi) -> None:
        """
        Menyisipkan lokasi baru berdasarkan kode_lokasi.
        Big-O: Rata-rata O(log n), Terburuk O(n).
        """
        if self.root is None:
            self.root = BSTNodeLok(lokasi)
        else:
            self._insert_recursive(self.root, lokasi)

    def _insert_recursive(self, current: BSTNodeLok, lokasi: Lokasi) -> None:
        if lokasi.kode < current.lokasi.kode:
            if current.left is None:
                current.left = BSTNodeLok(lokasi)
            else:
                self._insert_recursive(current.left, lokasi)
        elif lokasi.kode > current.lokasi.kode:
            if current.right is None:
                current.right = BSTNodeLok(lokasi)
            else:
                self._insert_recursive(current.right, lokasi)

    def search(self, kode: str) -> Optional[Lokasi]:
        """
        Mencari data lokasi berdasarkan kode unik.
        Big-O: Rata-rata O(log n), Terburuk O(n).
        """
        curr = self.root
        while curr:
            if kode == curr.lokasi.kode:
                return curr.lokasi
            curr = curr.left if kode < curr.lokasi.kode else curr.right
        return None

    def update_level(self, kode: str, level_baru: int) -> bool:
        """
        Mencari lokasi dan memperbarui tingkat keparahan bencana.
        Big-O: O(log n) rata-rata.
        """
        lokasi = self.search(kode)
        if lokasi:
            lokasi.level = level_baru
            return True
        return False

    def inorder(self) -> List[Lokasi]:
        """
        Menghasilkan daftar lokasi yang terurut secara alfabetis (kode).
        Big-O: O(n).
        """
        hasil = []
        self._inorder_recursive(self.root, hasil)
        return hasil

    def _inorder_recursive(self, current: Optional[BSTNodeLok], hasil: List[Lokasi]) -> None:
        if current:
            self._inorder_recursive(current.left, hasil)
            hasil.append(current.lokasi)
            self._inorder_recursive(current.right, hasil)
Penjelasan Program:
Struktur Data: Menggunakan dataclass untuk menyimpan profil lengkap lokasi bencana sesuai standar pendaftaran
.
Operasi insert: Menggunakan perbandingan string pada kode_lokasi untuk menentukan posisi node (kiri untuk kode lebih kecil, kanan untuk lebih besar)
.
Operasi search: Melakukan pencarian biner dengan efisiensi rata-rata O(log n), yang sangat krusial untuk validasi rute pada modul Dijkstra
.
Operasi update_level: Memanfaatkan fungsi search untuk menemukan data secara cepat sebelum melakukan pembaruan status urgensi di lapangan
.
Operasi inorder: Mengunjungi pohon dengan urutan Kiri-Akar-Kanan untuk mendapatkan daftar lokasi yang terurut rapi berdasarkan kodenya bagi kepentingan pelaporan
.
Program ini telah memenuhi syarat pengerjaan tanpa library luar dan memiliki analisis Big-O pada setiap operasi utamanya
.
